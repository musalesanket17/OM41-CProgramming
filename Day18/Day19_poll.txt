Q.3
int main(void) {
  char arr[4][8] = {"PG-DAC","PG-DESD","PG-DBDA"};
  printf("%c%s",**arr,*(arr+1)+1);
  return 0;
}
A) PPG-DAC
B) PPG-DESD
C) PPG-DBDA
D) PG-DESD
Answer: D
'P' --> %c -- **arr
"G-DESD" --> %s -- *(arr+1)+1
Q.4int main(void) {
  char arr[4][10]={"Sunbeam","Karad","Pune","Hinjewadi"};
  char *ptr = (char*)arr[3]; 
  *ptr++;
  printf("%s %s\n",arr[ptr - arr[3]],--ptr);
  return 0;
}
A. Sunbeam Hinjewadi
B. Sunbeam Pune
C. Compiler error
D. None of the above
Answer: A

--ptr --> "Hinjewadi"
ptr -> &arr[3]
ptr - arr[3] = 0
arr[0] = "Sunbeam"
