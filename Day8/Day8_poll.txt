5.#include <stdio.h>
int main()
{
    int a = 10, b = 5, c = 5, d;
    d = ~!b + b == c;
    printf("%d", !d);
    return 0;
}

A. 1
B. 0
C. 10
D. Syntax error

Answer: A

#include <stdio.h>
int main()
{
    int a = 10, b = 5, c = 5, d;
    d = ~!b + b == c;
    //  -1  + 5 == 5 
    //    4 == 5 
    //      0   
    printf("%d", !d); // !d --> !0 --> 1 
    return 0;
